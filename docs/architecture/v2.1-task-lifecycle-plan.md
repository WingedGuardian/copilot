        await self._log_event(task_id, "questions_set", questions[:200], db=db)

async def clear_pending_questions(self, task_id: str) -> None:
    """Clear pending questions and move task back to active."""
    async with aiosqlite.connect(self._db_path) as db:
        await db.execute(
            "UPDATE tasks SET pending_questions = NULL, status = 'active' WHERE id = ?",
            (task_id,),
        )
        await db.commit()
        await self._log_event(task_id, "questions_cleared", "", db=db)

async def get_tasks_with_questions(self) -> list[Task]:
    """Get tasks that have pending questions (status = awaiting)."""
    async with aiosqlite.connect(self._db_path) as db:
        db.row_factory = aiosqlite.Row
        cur = await db.execute(
            "SELECT * FROM tasks WHERE pending_questions IS NOT NULL AND status = 'awaiting'"
        )
        rows = await cur.fetchall()
        return [self._row_to_task(dict(r)) for r in rows]

async def add_steps_v2(self, task_id: str, steps: list[dict]) -> list[TaskStep]:

            "UPDATE tasks SET step_count = ? WHERE id = ?", (len(steps), task_id),

"""Task vs. conversation classifier — heuristic-first, no LLM call."""

from __future__ import annotations

import re
from dataclasses import dataclass


@dataclass
class DetectionResult:
    """Result of task detection."""
    is_task: bool
    confidence: float  # 0.0 to 1.0
    reason: str


# Imperative verbs that signal task intent
_IMPERATIVE_PATTERNS = re.compile(
    r"^(?:research|find|build|create|make|write|draft|set\s+up|compare"
    r"|analyze|look\s+(?:into|up)|check|send|schedule|organize|prepare"
    r"|update|fix|debug|deploy|configure|install|download|upload"
    r"|book|order|cancel|renew|register|sign\s+up|subscribe"
    r"|plan|design|implement|test|review|summarize|compile|generate"
    r"|get|fetch|pull|push|run|execute|handle|process|migrate)\b",
    re.I,
)

# Request patterns: "can you...", "I need...", "could you...", etc.
_REQUEST_PATTERNS = re.compile(
    r"^(?:can\s+you|could\s+you|would\s+you|please|i\s+need\s+(?:you\s+to|a)"
    r"|i(?:'d|\s+would)\s+like\s+(?:you\s+to|a)|i\s+want\s+(?:you\s+to|a)"
    r"|go\s+(?:ahead\s+and|and)|help\s+me)\b",
    re.I,
)

# Multi-step signal: "first...then", numbered lists
_MULTI_STEP_PATTERN = re.compile(
    r"(?:first.*then|step\s*\d|1\.|1\))", re.I | re.S,
)

# Deliverable keywords suggesting work product expected
_DELIVERABLE_PATTERNS = re.compile(
    r"\b(?:comparison\s+table|report|summary|list\s+of|spreadsheet"
    r"|draft|proposal|plan|analysis|breakdown|overview)\b",
    re.I,
)

# Short responses / acknowledgments — never tasks
_SHORT_ACK = re.compile(
    r"^(?:ok|okay|yes|no|yeah|nah|sure|thanks|thank\s+you|got\s+it"
    r"|cool|nice|great|good|perfect|awesome|lol|haha|interesting"
    r"|hmm|wow|right|yep|nope|k|ty|thx)\s*[.!?]*$",
    re.I,
)

MIN_TASK_LENGTH = 10  # Characters — shorter messages are never tasks


def classify_message(text: str) -> DetectionResult:
    """Classify a message as task request or conversation.

    Pure heuristic — no LLM call. Designed for speed (< 1ms).
    """
    text = text.strip()

    # Slash commands
    if text.startswith("/"):
        return DetectionResult(False, 0.95, "slash_command")

    # Too short
    if len(text) < MIN_TASK_LENGTH:
        return DetectionResult(False, 0.90, "too_short")

    # Short acknowledgments
    if _SHORT_ACK.match(text):
        return DetectionResult(False, 0.95, "acknowledgment")

    score = 0.0

    # Imperative verb at start of message
    if _IMPERATIVE_PATTERNS.match(text):
        score += 0.50

    # Request pattern
    if _REQUEST_PATTERNS.match(text):
        score += 0.40

    # Multi-step signals
    if _MULTI_STEP_PATTERN.search(text):
        score += 0.20

    # Deliverable keywords
    if _DELIVERABLE_PATTERNS.search(text):
        score += 0.20

    # Length bonus — longer messages more likely to be tasks
    if len(text) > 50:
        score += 0.10
    if len(text) > 150:
        score += 0.10

    # Cap at 1.0
    score = min(score, 1.0)

    is_task = score >= 0.40
    reason = "heuristic"
    if score >= 0.40:
        reason = "task_detected"

    return DetectionResult(is_task, score, reason)
```

    context_block = ""

   - "write" — creating text content, formatting, documents
   - "shell" — running commands, file operations
   - "code" — writing/modifying code (V2.2)
   - "browse" — interacting with websites (V2.3)
   - "general" — anything else
4. If you need more information to properly decompose the task, return clarifying_questions instead of steps

## Response Format (JSON only)
{{
  "steps": [
    {{"description": "Step description", "tool_type": "research|write|shell|code|browse|general"}}
  ],
  "clarifying_questions": ["question1", "question2"]

def build_progress_message(
    task_id: str,
    task_title: str,
    completed_steps: list[dict],
    current_step: dict | None = None,
    questions: list[str] | None = None,
) -> str:
    """Build a progress notification message for WhatsApp."""
    lines = [f"Task #{task_id}: {task_title}"]

    for step in completed_steps:
        lines.append(f"  done {step['description']}")

    if current_step:
        lines.append(f"  >> {current_step['description']}")

    if questions:
        lines.append("")
        for q in questions:
            lines.append(f"  ? {q}")

    return "\n".join(lines)

{task_summaries}

{user_message}

"""Task decomposition: LLM-powered step generation."""

from __future__ import annotations

import json
from dataclasses import dataclass, field

from loguru import logger


@dataclass
class DecompositionResult:
    """Result of decomposing a task into steps."""
    steps: list[dict] = field(default_factory=list)
    clarifying_questions: list[str] = field(default_factory=list)
    error: str | None = None


def parse_decomposition_response(raw: str) -> DecompositionResult:
    """Parse the LLM's decomposition response into structured data."""
    text = raw.strip()

    # Strip markdown fences
    if text.startswith("```"):
        text = text.split("\n", 1)[-1].rsplit("```", 1)[0].strip()

    try:
        data = json.loads(text)
    except json.JSONDecodeError as e:
        logger.warning(f"Decomposition parse failed: {e}")
        return DecompositionResult(error=f"Invalid JSON: {e}")

    steps = data.get("steps", [])
    questions = data.get("clarifying_questions", [])

    # Validate steps have required fields
    valid_steps = []
    for step in steps:
        if isinstance(step, dict) and "description" in step:
            valid_steps.append({
                "description": step["description"],
                "tool_type": step.get("tool_type", "general"),
            })

    return DecompositionResult(steps=valid_steps, clarifying_questions=questions)
```

```python
"""Task worker: background executor for persistent task queue."""

from __future__ import annotations

import asyncio
from typing import Any, Callable, Awaitable

from loguru import logger

from nanobot.copilot.tasks.manager import TaskManager
from nanobot.copilot.tasks.prompts import build_progress_message


class TaskWorker:
    """Background worker that picks up and executes pending tasks.

    V2.1 enhancements:
    - decompose_fn returns structured steps (list[dict]) or questions (dict)
    - execute_fn receives tool_type for dispatch routing
    - notify_fn sends progress updates to user
    """

    def __init__(
        self,
        task_manager: TaskManager,
        execute_fn: Callable[[str, str, str, str], Awaitable[str]],
        decompose_fn: Callable[[str], Awaitable[list[dict] | dict]] | None = None,
        notify_fn: Callable[[str], Awaitable[None]] | None = None,
        interval_s: int = 60,
    ):
        self._manager = task_manager
        self._execute_fn = execute_fn
        self._decompose_fn = decompose_fn
        self._notify_fn = notify_fn
        self._interval = interval_s
        self._running = False
        self._task: asyncio.Task | None = None

    async def start(self) -> None:
        """Start the background worker loop."""
        self._running = True
        self._task = asyncio.create_task(self._loop())
        logger.info(f"Task worker started (interval={self._interval}s)")

    def stop(self) -> None:
        """Stop the worker."""
        self._running = False
        if self._task:
            self._task.cancel()

    async def _loop(self) -> None:
        """Main worker loop."""
        while self._running:
            try:
                await self._tick()
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Task worker tick failed: {e}")
            await asyncio.sleep(self._interval)

    async def _tick(self) -> None:
        """Process one pending task."""
        task = await self._manager.get_next_pending()
        if not task:
            return

        logger.info(f"Task worker picked up: {task.id} - {task.title}")
        await self._manager.update_status(task.id, "active")

        # If no steps, try decomposition
        if task.step_count == 0:
            if self._decompose_fn:
                await self._decompose_task(task)
                return  # Will pick up steps on next tick
            else:
                # No decomposer — execute whole task directly
                await self._execute_whole_task(task)
                return

        # Execute next step
        await self._execute_next_step(task)

    async def _decompose_task(self, task) -> None:
        """Decompose a task into steps using the frontier model."""
        try:
            result = await self._decompose_fn(task.description or task.title)

            # Handle dict response (may contain clarifying_questions)
            if isinstance(result, dict):
                questions = result.get("clarifying_questions", [])
                steps = result.get("steps", [])
                if questions and not steps:
                    # Need clarification — pause task
                    q_text = "\n".join(f"- {q}" for q in questions)
                    await self._manager.set_pending_questions(task.id, q_text)
                    await self._notify(
                        f"Task #{task.id}: {task.title}\n\n"
                        f"I need some clarification before I can proceed:\n{q_text}"
                    )
                    logger.info(f"Task {task.id} awaiting clarification")
                    return
                elif steps:
                    result = steps

            # result is list[dict] at this point
            if result:
                await self._manager.add_steps_v2(task.id, result)
                task.step_count = len(result)
                logger.info(f"Task {task.id} decomposed into {len(result)} steps")
            else:
                logger.warning(f"Task {task.id} decomposition returned empty")
                await self._manager.update_status(task.id, "failed")

        except Exception as e:
            logger.warning(f"Task decomposition failed for {task.id}: {e}")
            await self._manager.update_status(task.id, "failed")
            await self._notify(f"Task #{task.id}: {task.title}\n\nDecomposition failed: {e}")

    async def _execute_next_step(self, task) -> None:
        """Execute the next pending step of a task."""
        step = await self._manager.get_next_step(task.id)
        if not step:
            # All steps done
            await self._manager.complete_task(task.id)
            await self._notify_completion(task)
            logger.info(f"Task completed: {task.id}")
            return

        tool_type = getattr(step, "tool_type", "general")

        try:
            result = await self._execute_fn(
                step.description,
                task.session_key or f"task:{task.id}",
                "cli",
                tool_type,
            )
            await self._manager.complete_step(task.id, step.step_index, result[:2000])

            # Send progress notification
            await self._notify_progress(task, step)

        except Exception as e:
            await self._manager.fail_step(task.id, step.step_index, str(e))
            logger.error(f"Task step failed: {task.id}/{step.step_index}: {e}")
            await self._notify(
                f"Task #{task.id}: {task.title}\n\n"
                f"Step {step.step_index + 1} failed: {step.description}\n"
                f"Error: {e}"
            )

    async def _execute_whole_task(self, task) -> None:
        """Execute a task without decomposition (direct execution)."""
        try:
            result = await self._execute_fn(
                task.description or task.title,
                task.session_key or f"task:{task.id}",
                "cli",
                "general",
            )
            await self._manager.complete_task(task.id)
            await self._notify(
                f"Task #{task.id}: {task.title}\n\n"
                f"Completed:\n{result[:1500]}"
            )
            logger.info(f"Task completed (no decomposition): {task.id}")
        except Exception as e:
            await self._manager.update_status(task.id, "failed")
            logger.error(f"Task execution failed: {task.id}: {e}")

    async def _notify_progress(self, task, completed_step) -> None:
        """Send a progress notification after a step completes."""
        # Reload task to get current state
        updated = await self._manager.get_task(task.id)
        if not updated:
            return

        completed = [s for s in updated.steps if s.status == "completed"]
        pending = [s for s in updated.steps if s.status == "pending"]

        msg = build_progress_message(
            task_id=task.id,
            task_title=task.title,
            completed_steps=[{"description": s.description} for s in completed],
            current_step={"description": pending[0].description} if pending else None,
        )
        await self._notify(msg)

    async def _notify_completion(self, task) -> None:
        """Send a completion notification."""
        updated = await self._manager.get_task(task.id)
        if not updated:
            return

        # Collect results from all steps
        results = []
        for step in updated.steps:
            if step.result:
                results.append(step.result)

        final_result = "\n\n".join(results) if results else "(no output)"
        # Truncate for WhatsApp
        if len(final_result) > 3000:
            final_result = final_result[:3000] + "\n\n... (truncated)"

        await self._notify(
            f"Task #{task.id} complete: {task.title}\n\n{final_result}"
        )

    async def _notify(self, message: str) -> None:
        """Send a notification if notify_fn is configured."""
        if self._notify_fn:
            try:
                await self._notify_fn(message)
            except Exception as e:
                logger.warning(f"Task notification failed: {e}")
```

"""Feedback handler: route user replies to blocked tasks.

Uses simple heuristics for V2.1. Can be upgraded to LLM-based resolution
in V2.2 when the coordinator model role is formalized.
"""

from __future__ import annotations

import re
from dataclasses import dataclass

from loguru import logger


@dataclass
class FeedbackResult:
    """Result of feedback resolution."""
    matches_task: str | None = None  # task_id or None
    answer_text: str | None = None
    is_new_task: bool = False
    reasoning: str = ""


# Patterns that indicate conversational intent, not task answers
_CONVERSATIONAL_PATTERNS = re.compile(
    r"^(?:hi|hello|hey|how\s+are|what'?s\s+up|good\s+morning"
    r"|good\s+afternoon|good\s+evening|thanks|thank\s+you"
    r"|help|status|what\s+can\s+you)\b",
    re.I,
)


def resolve_feedback(
    user_message: str,
    pending_tasks: list[dict],
) -> FeedbackResult:
    """Determine if a user message answers a pending task question.

    Heuristic approach for V2.1:
    - If no pending tasks, no match
    - If one pending task, bias toward matching (user likely responding)
    - If conversational pattern, no match
    - Keyword overlap between message and question increases match score

    Args:
        user_message: The user's reply text
        pending_tasks: List of dicts with 'id', 'title', 'questions' keys
    """
    if not pending_tasks:
        return FeedbackResult(reasoning="no_pending_tasks")

    text = user_message.strip()

    # Skip obvious conversational messages
    if _CONVERSATIONAL_PATTERNS.match(text):
        return FeedbackResult(reasoning="conversational_pattern")

    # Skip slash commands
    if text.startswith("/"):
        return FeedbackResult(reasoning="slash_command")

    # Single pending task — bias toward matching
    if len(pending_tasks) == 1:
        task = pending_tasks[0]
        # Only skip if message is very short AND looks like a greeting
        if len(text) < 5 and not any(c.isdigit() for c in text):
            return FeedbackResult(reasoning="too_short_ambiguous")

        return FeedbackResult(
            matches_task=task["id"],
            answer_text=text,
            reasoning="single_task_bias",
        )

    # Multiple pending tasks — keyword overlap scoring
    best_score = 0.0
    best_task = None

    for task in pending_tasks:
        questions = task.get("questions", "").lower()
        msg_lower = text.lower()

        # Word overlap between message and question
        q_words = set(re.findall(r"\w{3,}", questions))
        m_words = set(re.findall(r"\w{3,}", msg_lower))

        if not q_words:
            continue

        overlap = len(q_words & m_words)
        score = overlap / len(q_words) if q_words else 0

        # Boost if task title words appear in message
        title_words = set(re.findall(r"\w{3,}", task.get("title", "").lower()))
        title_overlap = len(title_words & m_words)
        score += title_overlap * 0.1

        if score > best_score:
            best_score = score
            best_task = task

    if best_task and best_score > 0.1:
        return FeedbackResult(
            matches_task=best_task["id"],
            answer_text=text,
            reasoning=f"keyword_overlap_{best_score:.2f}",
        )

    return FeedbackResult(reasoning="no_match")
```

                        lines.append(f"  {icon} {s.step_index + 1}. {s.description}{result_preview}")
                return OutboundMessage(channel=msg.channel, chat_id=msg.chat_id,
                                      content="\n".join(lines))
            return OutboundMessage(channel=msg.channel, chat_id=msg.chat_id,
                                  content="Task system not available.")

        if cmd.startswith("/cancel "):
            task_id = cmd[8:].strip()
            if self._task_manager:
                task = await self._task_manager.get_task(task_id)
                if not task:
                    return OutboundMessage(channel=msg.channel, chat_id=msg.chat_id,
                                          content=f"Task #{task_id} not found.")
                await self._task_manager.update_status(task_id, "failed")
                return OutboundMessage(channel=msg.channel, chat_id=msg.chat_id,
                                      content=f"Task #{task_id} cancelled.")
            return OutboundMessage(channel=msg.channel, chat_id=msg.chat_id,
                                  content="Task system not available.")
```
